void RobotControl(){
	if(an[0] <= black_threshold && an[1] > black_threshold)   // WHITE, BLACK
	{
		Set_PWM(0, 350);  // Motor 0 fast
		Set_PWM(1, 50);  // Motor 1 slow
	}
	else 
	if(an[0] > black_threshold && an[1] <= black_threshold)   // BLACK, WHITE
	{
		Set_PWM(0, 50);  // Motor 0 slow
		Set_PWM(1, 350);  // Motor 1 fast
	}
	else
	if(an[0] > black_threshold && an[1] > black_threshold)    // BLACK, BLACK
	{
		Set_PWM(0, 200);  // Motor 0 medium
		Set_PWM(1, 200);  // Motor 1 medium
	}		
}
